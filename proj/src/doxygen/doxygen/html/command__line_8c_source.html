102void translate_scancode(uint8_t scancode) {
103 char ch;
104 switch (scancode) {
105 case WAITING_FOR_ANOTHER_CHAR:
106 waiting_for_another_char = true;
107 break;
108 case LEFT_SHIFT_MAKE_CODE:
109 left_shift_pressed = true;
110 break;
111 case RIGHT_SHIFT_MAKE_CODE:
112 right_shift_pressed = true;
113 break;
114 case LEFT_SHIFT_BREAK_CODE:
115 left_shift_pressed = false;
116 break;
117 case RIGHT_SHIFT_BREAK_CODE:
118 right_shift_pressed = false;
119 break;
120 case CAPS_LOCK_MAKE_CODE:
121 caps_lock = !caps_lock;
122 break;
123 case BACKSPACE_MAKE_CODE:
124 delete_character();
125 break;
126 default:
127 if (waiting_for_another_char) {
128 waiting_for_another_char = false;
129 switch (scancode) {
130 case ALTGR_MAKE_CODE:
131 altgr_pressed = true;
132 break;
133 case ALTGR_BREAK_CODE:
134 altgr_pressed = false;
135 break;
136 case LEFT_ARROW_MAKE_CODE: // Left arrow key
137 if (cursor_pos > 0) {
138 printf("\033[D"); // Move cursor left
139 cursor_pos--;
140 }
141 break;
142 case RIGHT_ARROW_MAKE_CODE: // Right arrow key
143 if (cursor_pos < line_length) {
144 printf("\033[C"); // Move cursor right
145 cursor_pos++;
146 }
147 break;
148 case DOWN_ARROW_MAKE_CODE: // Down arrow key (not handled)
149 break;
150 case UP_ARROW_MAKE_CODE: // Up arrow key (not handled)
151 break;
152 case DELETE_MAKE_CODE:
153 delete_character();
154 }
155 break;
156 }
157 if ((scancode & BIT(7)) == 0) {
158 ch = scancode_to_char[scancode];
159 if (caps_lock && ch >= 'a' && ch <= 'z') {
160 if (!left_shift_pressed && !right_shift_pressed)
161 ch -= 32;
162 } else if (left_shift_pressed || right_shift_pressed) {
163 char char_to_print = shift_scancode_to_char[scancode];
164 if (char_to_print)
165 ch = char_to_print;
166 } else if (altgr_pressed) {
167 char char_to_print = altgr_scancode_to_char[scancode];
168 if (char_to_print)
169 ch = char_to_print;
170 }
171
172 if (ch) {
173 if (cursor_pos < line_length)
174 memmove(&line_buffer[cursor_pos + 1], &line_buffer[cursor_pos],
175 line_length - cursor_pos);
176 line_buffer[cursor_pos] = ch;
177 line_length++;
178 line_buffer[line_length] = '\0';
179 printf("%c", ch);
180 if (cursor_pos < line_length) {
181 printf("%s", &line_buffer[cursor_pos + 1]);
182 for (int i = line_length; i > cursor_pos + 1; i--) {
183 printf("\033[D");
184 }
185 }
186 cursor_pos++;
187 }
188 }
189 }

204void tokenize_line(char* line_buffer) {
205
206 memset(tokens, 0, sizeof(tokens));
207
208 char *token = strtok(line_buffer, " ");
209 int index = 0;
210 while (token != NULL && index < MAX_TOKENS - 1) {
211 tokens[index++] = token;
212 token = strtok(NULL, " ");
213 }
214
215 tokens[index] = NULL;
216}
