116int (project_loop)() {
117 int r;
118
119 if(rtc_get_time() != 0) return 1;
120
121 if(draw_page() != 0) {
122 printf("Error: Problems occured while trying to draw the main menu! \n");
123 return 1;
124 }
125
126 int ipc_status;
127 message msg;
128
129 while (running) {
130 if( (r = driver_receive(ANY, &msg, &ipc_status)) != 0 ) {
131 printf("Error %s\r", r);
132 continue;
133 }
134
135 if(is_ipc_notify(ipc_status)) {
136 switch(_ENDPOINT_P(msg.m_source)){
137 case HARDWARE:
138 if (msg.m_notify.interrupts & irq_timer) {
139
140 timer_handler[page_state].handler();
141 }
142
143 if (msg.m_notify.interrupts & irq_keyboard) {
144 keyboard_handler[page_state].handler();
145 }
146
147 if (msg.m_notify.interrupts & irq_mouse) {
148 mouse_handler[page_state].handler();
149 }
150
151 if (msg.m_notify.interrupts & irq_real_time_clock) {
152 real_time_clock_handler[page_state].handler();
153 }
154
155 if (msg.m_notify.interrupts & irq_serial_port) {
156 serial_port_handler[page_state].handler(serial_port_handler);
157 }
158
159 break;
160 default:
161 break;
162 }
163 }
164 }
165
166 return 0;
167}
